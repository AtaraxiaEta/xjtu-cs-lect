\chapter{母函数与递推关系}

\section{母函数}

    \begin{definition}
        [(普)母函数]
        对于序列$a_0,a_1,a_2,\cdots,a_n,\cdots$构造一函数$G(x)=a_0+a_1x+a_2x^2+\cdots+a_nx^n\cdots$, 称函数$G(x)$是序列$a_0,a_1,a_2,\cdots,a_n,\cdots$上的\textsf{(普)母函数}.
    \end{definition}
